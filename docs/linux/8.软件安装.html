# 1.包管理器

  • centos，yum包管理器，格式rpm
  • ubuntu，apt包管理器，格式deb
# rpm包格式

软件名称 软件版本 系统版本 平台

  • -q查询软件包
  • -i安装软件包
  • -e卸载软件包
ls /dev/sr0 -l
//光驱安装
dd if=/dev/sr0  of=/xx/xx.iso
//u盘挂载
mount /dev/sr0 /mnt

rpm -q vim-common
rpm -i vim-common=7.4.160-5.el7.X86_64.rpm
rpm -e vim-common
